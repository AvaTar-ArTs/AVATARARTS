### Revenue Asset Scanner
```python
#!/usr/bin/env python3
"""
Revenue Asset Scanner for AVATARARTS Migration
Scans the entire AVATARARTS ecosystem for revenue-generating assets
"""

import os
import json
from pathlib import Path
from datetime import datetime
import csv

def scan_revenue_assets(base_path="/Users/steven/AVATARARTS"):
    """Scan for revenue-generating assets"""
    base_dir = Path(base_path)
    assets = {
        "python_scripts": [],
        "ai_content": [],
        "automation_workflows": [],
        "product_packages": [],
        "revenue_systems": [],
        "seo_tools": [],
        "marketplace_assets": []
    }
    
    # Scan for Python scripts
    for py_file in base_dir.rglob("*.py"):
        if py_file.stat().st_size > 0:  # Non-empty files
            content = py_file.read_text(errors='ignore')
            # Look for revenue-related keywords
            if any(keyword in content.lower() for keyword in [
                'revenue', 'income', 'sales', 'profit', 'monetize', 'payment', 
                'gumroad', 'codecanyon', 'etsy', 'upwork', 'fiverr', 'apify',
                'automation', 'ai', 'content', 'marketing', 'seo', 'aeo'
            ]):
                assets["python_scripts"].append({
                    "path": str(py_file.relative_to(base_dir)),
                    "size": py_file.stat().st_size,
                    "modified": datetime.fromtimestamp(py_file.stat().st_mtime).isoformat(),
                    "contains_revenue_keywords": True
                })
    
    # Scan for AI content (videos, images, audio)
    for media_file in base_dir.rglob("*"):
        if media_file.is_file():
            ext = media_file.suffix.lower()
            if ext in ['.mp4', '.mp3', '.wav', '.mov', '.avi', '.mkv', '.m4a']:
                # Look for AI-related content
                if any(keyword in str(media_file).lower() for keyword in [
                    'ai', 'sora', 'runway', 'veo', 'pika', 'kling', 'midjourney', 
                    'dalle', 'stability', 'video', 'music', 'audio', 'image'
                ]):
                    assets["ai_content"].append({
                        "path": str(media_file.relative_to(base_dir)),
                        "size": media_file.stat().st_size,
                        "modified": datetime.fromtimestamp(media_file.stat().st_mtime).isoformat(),
                        "type": ext
                    })
    
    # Scan for automation workflows
    for workflow_file in base_dir.rglob("*.json"):
        if 'n8n' in str(workflow_file) or 'workflow' in str(workflow_file).lower():
            assets["automation_workflows"].append({
                "path": str(workflow_file.relative_to(base_dir)),
                "size": workflow_file.stat().st_size,
                "modified": datetime.fromtimestamp(workflow_file.stat().st_mtime).isoformat()
            })
    
    # Scan for product packages
    for zip_file in base_dir.rglob("*.zip"):
        if any(keyword in str(zip_file).lower() for keyword in [
            'bundle', 'package', 'product', 'gumroad', 'codecanyon', 'etsy'
        ]):
            assets["product_packages"].append({
                "path": str(zip_file.relative_to(base_dir)),
                "size": zip_file.stat().st_size,
                "modified": datetime.fromtimestamp(zip_file.stat().st_mtime).isoformat()
            })
    
    # Scan for revenue systems
    for revenue_file in base_dir.rglob("*"):
        if revenue_file.is_file():
            name_lower = str(revenue_file).lower()
            if any(keyword in name_lower for keyword in [
                'revenue', 'dashboard', 'tracking', 'analytics', 'roi', 'profit'
            ]):
                assets["revenue_systems"].append({
                    "path": str(revenue_file.relative_to(base_dir)),
                    "size": revenue_file.stat().st_size,
                    "modified": datetime.fromtimestamp(revenue_file.stat().st_mtime).isoformat()
                })
    
    return assets

def generate_migration_report(assets, output_dir):
    """Generate comprehensive migration report"""
    output_path = Path(output_dir) / "revenue_asset_inventory.json"
    
    report = {
        "scan_date": datetime.now().isoformat(),
        "base_path": "/Users/steven/AVATARARTS",
        "assets_found": {
            "python_scripts": len(assets["python_scripts"]),
            "ai_content": len(assets["ai_content"]),
            "automation_workflows": len(assets["automation_workflows"]),
            "product_packages": len(assets["product_packages"]),
            "revenue_systems": len(assets["revenue_systems"]),
            "seo_tools": len(assets["seo_tools"]),
            "marketplace_assets": len(assets["marketplace_assets"])
        },
        "detailed_assets": assets,
        "migration_recommendations": []
    }
    
    # Add recommendations based on findings
    if assets["python_scripts"]:
        report["migration_recommendations"].append({
            "category": "python_scripts",
            "action": "Migrate to 02_CORE_ASSETS/automation_scripts/",
            "priority": "high",
            "estimated_time": "2-4 hours"
        })
    
    if assets["ai_content"]:
        report["migration_recommendations"].append({
            "category": "ai_content", 
            "action": "Migrate to 02_CORE_ASSETS/ai_video_generation/",
            "priority": "high", 
            "estimated_time": "1-2 hours"
        })
    
    if assets["product_packages"]:
        report["migration_recommendations"].append({
            "category": "product_packages",
            "action": "Migrate to 03_MONETIZATION/product_packages/",
            "priority": "high",
            "estimated_time": "30-60 minutes"
        })
    
    with open(output_path, 'w') as f:
        json.dump(report, f, indent=2)
    
    print(f"üìä Revenue asset inventory saved to: {output_path}")
    print(f"   Total assets identified: {sum(report['assets_found'].values())}")
    print(f"   Python scripts: {report['assets_found']['python_scripts']}")
    print(f"   AI content: {report['assets_found']['ai_content']}")
