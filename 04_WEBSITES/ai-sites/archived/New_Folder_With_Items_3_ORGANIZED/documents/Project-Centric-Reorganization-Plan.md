```python
#!/usr/bin/env python3
"""
Project-Centric Migration Script
Automatically moves files to project-based structure
"""

import os
import shutil
from pathlib import Path

# Project mapping
PROJECT_MAPPING = {
    "01_AI_Tools_and_Projects": "01_ACTIVE_PROJECTS/AI_Automation_System",
    "aGPT": "01_ACTIVE_PROJECTS/AI_Automation_System",
    "DeepSeek": "01_ACTIVE_PROJECTS/AI_Automation_System",
    "Quantum AI & Automation": "01_ACTIVE_PROJECTS/AI_Automation_System",
    "Whisper-Text": "01_ACTIVE_PROJECTS/AI_Automation_System",
    "03_Creative_Projects": "01_ACTIVE_PROJECTS/Creative_Design_Portfolio",
    "comic": "01_ACTIVE_PROJECTS/Creative_Design_Portfolio",
    "Adobe": "01_ACTIVE_PROJECTS/Creative_Design_Portfolio",
    "04_Development_and_Code": "01_ACTIVE_PROJECTS/Web_Development_Suite",
    "code": "01_ACTIVE_PROJECTS/Web_Development_Suite",
    "github": "01_ACTIVE_PROJECTS/Web_Development_Suite",
    "python": "01_ACTIVE_PROJECTS/Web_Development_Suite",
    "script": "01_ACTIVE_PROJECTS/Web_Development_Suite",
    "02_Business_and_Finance": "01_ACTIVE_PROJECTS/Business_Ventures",
    "SEO_MARKETING_STRATEGY": "01_ACTIVE_PROJECTS/Business_Ventures",
    "05_Documentation_and_Notes": "02_REFERENCE_PROJECTS/Documentation_Library",
    "documentation": "02_REFERENCE_PROJECTS/Documentation_Library",
    "notes": "02_REFERENCE_PROJECTS/Documentation_Library",
    "Obsidian Vault": "02_REFERENCE_PROJECTS/Documentation_Library",
    "06_Archives_and_Backups": "03_ARCHIVED_PROJECTS/Legacy_Projects",
    "duplicates": "03_ARCHIVED_PROJECTS/Legacy_Projects",
    "misc": "03_ARCHIVED_PROJECTS/Legacy_Projects",
    "trivia": "03_ARCHIVED_PROJECTS/Legacy_Projects"
}

def migrate_projects():
    """Migrate files to project-based structure"""
    documents_path = Path.home() / "Documents"
    
    for source, destination in PROJECT_MAPPING.items():
        source_path = documents_path / source
        dest_path = documents_path / destination
        
        if source_path.exists():
            print(f"Moving {source} to {destination}")
            dest_path.mkdir(parents=True, exist_ok=True)
            shutil.move(str(source_path), str(dest_path))
        else:
            print(f"Source {source} not found, skipping")

if __name__ == "__main__":
    migrate_projects()
```
