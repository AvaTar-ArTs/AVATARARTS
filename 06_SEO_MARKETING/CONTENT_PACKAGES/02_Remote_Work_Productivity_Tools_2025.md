### Remote Work Time Tracker (Python)

```python
"""
Remote Work Time Tracker
Tracks work hours, breaks, and productivity metrics
"""

from datetime import datetime, timedelta
import json

class RemoteWorkTracker:
    def __init__(self):
        self.sessions = []
        self.breaks = []
        
    def start_work_session(self, task_name, project=None):
        """Start a work session"""
        session = {
            'task': task_name,
            'project': project,
            'start_time': datetime.now().isoformat(),
            'end_time': None,
            'duration': None
        }
        self.sessions.append(session)
        return session
    
    def end_work_session(self, session_index):
        """End a work session"""
        if session_index < len(self.sessions):
            session = self.sessions[session_index]
            end_time = datetime.now()
            start_time = datetime.fromisoformat(session['start_time'])
            duration = end_time - start_time
            
            session['end_time'] = end_time.isoformat()
            session['duration'] = str(duration)
            return session
        return None
    
    def take_break(self, break_type='short', duration_minutes=5):
        """Record a break"""
        break_record = {
            'type': break_type,
            'start_time': datetime.now().isoformat(),
            'duration_minutes': duration_minutes
        }
        self.breaks.append(break_record)
        return break_record
    
    def get_daily_stats(self, date=None):
        """Get daily productivity statistics"""
        if date is None:
            date = datetime.now().date()
        
        daily_sessions = [
            s for s in self.sessions 
            if datetime.fromisoformat(s['start_time']).date() == date
        ]
        
        total_time = timedelta()
        for session in daily_sessions:
            if session['duration']:
                hours, minutes, seconds = map(int, session['duration'].split(':'))
                total_time += timedelta(hours=hours, minutes=minutes, seconds=seconds)
        
        return {
            'date': date.isoformat(),
            'sessions': len(daily_sessions),
            'total_work_time': str(total_time),
            'breaks': len([b for b in self.breaks if datetime.fromisoformat(b['start_time']).date() == date])
        }
    
    def export_data(self, filename='work_tracker.json'):
        """Export tracking data to JSON"""
        data = {
            'sessions': self.sessions,
            'breaks': self.breaks
        }
        with open(filename, 'w') as f:
            json.dump(data, f, indent=2)
        return filename

# Example usage
tracker = RemoteWorkTracker()

# Start work sessions
tracker.start_work_session('Project Planning', 'Q1 Planning')
tracker.start_work_session('Code Review', 'Product Development')

# Take breaks
tracker.take_break('short', 5)
tracker.take_break('lunch', 30)

# Get daily stats
stats = tracker.get_daily_stats()
print(f"Today's Sessions: {stats['sessions']}")
print(f"Total Work Time: {stats['total_work_time']}")
```

### Productivity Dashboard (JavaScript)

```javascript
/**
 * Remote Work Productivity Dashboard
 * Tracks and visualizes productivity metrics
 */

class ProductivityDashboard {
  constructor() {
    this.tasks = [];
    this.focusSessions = [];
    this.distractions = [];
  }

  addTask(task) {
    this.tasks.push({
      ...task,
      id: Date.now(),
      createdAt: new Date().toISOString(),
      completed: false
    });
  }

  completeTask(taskId) {
    const task = this.tasks.find(t => t.id === taskId);
    if (task) {
      task.completed = true;
      task.completedAt = new Date().toISOString();
    }
  }

  startFocusSession(duration = 25) {
    const session = {
      id: Date.now(),
      startTime: new Date().toISOString(),
      duration: duration,
      completed: false
    };
    this.focusSessions.push(session);
    return session;
  }

  recordDistraction(type, duration) {
    this.distractions.push({
      id: Date.now(),
      type: type,
      duration: duration,
      timestamp: new Date().toISOString()
    });
  }

  getProductivityScore() {
    const completedTasks = this.tasks.filter(t => t.completed).length;
    const totalTasks = this.tasks.length;
    const taskCompletionRate = totalTasks > 0 ? completedTasks / totalTasks : 0;

    const completedSessions = this.focusSessions.filter(s => s.completed).length;
    const totalSessions = this.focusSessions.length;
    const focusRate = totalSessions > 0 ? completedSessions / totalSessions : 0;

    const distractionCount = this.distractions.length;
    const distractionPenalty = Math.min(distractionCount * 0.1, 0.5);

    const score = (taskCompletionRate * 0.5 + focusRate * 0.5) * (1 - distractionPenalty);
    return Math.round(score * 100);
  }

  getDailyReport() {
    const today = new Date().toDateString();
    const todayTasks = this.tasks.filter(t => 
      new Date(t.createdAt).toDateString() === today
    );
    const todaySessions = this.focusSessions.filter(s =>
      new Date(s.startTime).toDateString() === today
    );

    return {
      date: today,
      tasksCompleted: todayTasks.filter(t => t.completed).length,
      totalTasks: todayTasks.length,
      focusSessions: todaySessions.length,
      distractions: this.distractions.length,
      productivityScore: this.getProductivityScore()
    };
  }
}

// Example usage
const dashboard = new ProductivityDashboard();

dashboard.addTask({ title: 'Review project proposal', priority: 'high' });
dashboard.addTask({ title: 'Update documentation', priority: 'medium' });
dashboard.completeTask(dashboard.tasks[0].id);

dashboard.startFocusSession(25);
dashboard.recordDistraction('social media', 10);

const report = dashboard.getDailyReport();
console.log('Daily Report:', report);
console.log('Productivity Score:', dashboard.getProductivityScore());
```
