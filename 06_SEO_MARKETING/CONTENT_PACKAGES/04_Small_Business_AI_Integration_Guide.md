"""
Small Business AI Integration Assessment
Evaluates readiness and recommends tools
"""

class AIIntegrationAssessment:
    def __init__(self):
        self.questions = {
            'customer_service': {
                'volume': 'How many customer inquiries do you receive daily?',
                'hours': 'How many hours per week spent on customer service?',
                'satisfaction': 'Current customer satisfaction score (1-10)?'
            },
            'marketing': {
                'campaigns': 'How many marketing campaigns per month?',
                'automation': 'Current marketing automation level (1-10)?',
                'leads': 'Monthly lead generation count?'
            },
            'sales': {
                'pipeline': 'Number of deals in pipeline?',
                'conversion': 'Current conversion rate (%)?',
                'followup': 'Hours per week on follow-ups?'
            },
            'operations': {
                'repetitive': 'Hours per week on repetitive tasks?',
                'errors': 'Error rate in key processes (%)?',
                'efficiency': 'Current efficiency score (1-10)?'
            }
        }
        
    def assess_area(self, area, responses):
        """Assess a specific business area"""
        score = 0
        recommendations = []
        
        if area == 'customer_service':
            if responses['volume'] > 50:
                score += 3
                recommendations.append('High inquiry volume - chatbot recommended')
            if responses['hours'] > 10:
                score += 3
                recommendations.append('Significant time spent - automation needed')
            if responses['satisfaction'] < 7:
                score += 2
                recommendations.append('Low satisfaction - AI support can help')
                
        elif area == 'marketing':
            if responses['campaigns'] > 5:
                score += 3
                recommendations.append('Multiple campaigns - automation recommended')
            if responses['automation'] < 5:
                score += 3
                recommendations.append('Low automation - high potential')
            if responses['leads'] > 100:
                score += 2
                recommendations.append('High lead volume - CRM needed')
                
        elif area == 'sales':
            if responses['pipeline'] > 20:
                score += 3
                recommendations.append('Large pipeline - CRM essential')
            if responses['conversion'] < 20:
                score += 3
                recommendations.append('Low conversion - lead scoring needed')
            if responses['followup'] > 5:
                score += 2
                recommendations.append('Time on follow-ups - automation recommended')
                
        elif area == 'operations':
            if responses['repetitive'] > 10:
                score += 3
                recommendations.append('High repetitive tasks - automation priority')
            if responses['errors'] > 5:
                score += 3
                recommendations.append('High error rate - AI validation needed')
            if responses['efficiency'] < 6:
                score += 2
                recommendations.append('Low efficiency - process optimization needed')
        
        return {
            'score': min(score, 10),
            'priority': 'high' if score >= 7 else 'medium' if score >= 4 else 'low',
            'recommendations': recommendations
        }
    
    def generate_roadmap(self, assessments):
        """Generate implementation roadmap"""
        # Sort by priority and score
        sorted_areas = sorted(
            assessments.items(),
            key=lambda x: (x[1]['priority'] == 'high', x[1]['score']),
            reverse=True
        )
        
        roadmap = {
            'phase1_quick_wins': [],
            'phase2_core': [],
            'phase3_advanced': []
        }
        
        for area, assessment in sorted_areas:
            if assessment['priority'] == 'high' and assessment['score'] >= 7:
                roadmap['phase1_quick_wins'].append({
                    'area': area,
                    'assessment': assessment
                })
            elif assessment['priority'] == 'high':
                roadmap['phase2_core'].append({
                    'area': area,
                    'assessment': assessment
                })
            else:
                roadmap['phase3_advanced'].append({
                    'area': area,
                    'assessment': assessment
                })
        
        return roadmap

# Example usage
assessment = AIIntegrationAssessment()

# Assess customer service

    'hours': 15,
    'satisfaction': 6
}

/**
 * AI Tool ROI Calculator for Small Businesses
 * Calculates return on investment for AI implementations
 */

class AIROICalculator {
  constructor() {
    this.tools = [];
    this.metrics = {
      timeSaved: 0,
      costReduction: 0,
      revenueIncrease: 0
    };
  }

  addTool(tool) {
    this.tools.push({
      ...tool,
      id: Date.now(),
      addedAt: new Date().toISOString()
    });
    this.calculateROI();
  }

  calculateTimeSavings() {
    return this.tools.reduce((total, tool) => {
      return total + (tool.hoursSavedPerWeek || 0);
    }, 0);
  }

  calculateCostSavings(hourlyRate = 25) {
    const hoursSaved = this.calculateTimeSavings();
    const weeklySavings = hoursSaved * hourlyRate;
    const monthlySavings = weeklySavings * 4.33; // Average weeks per month
    return {
      weekly: weeklySavings,
      monthly: monthlySavings,
      yearly: monthlySavings * 12
    };
  }

  calculateToolCosts() {
    return this.tools.reduce((total, tool) => {
      return total + (tool.monthlyCost || 0);
    }, 0);
  }

  calculateRevenueImpact(currentRevenue, growthRate) {
    const monthlyIncrease = currentRevenue * (growthRate / 100);
    return {
      monthly: monthlyIncrease,
      yearly: monthlyIncrease * 12
    };
  }

  calculateROI(currentRevenue = 10000, hourlyRate = 25, growthRate = 15) {
    const timeSavings = this.calculateTimeSavings();
    const costSavings = this.calculateCostSavings(hourlyRate);
    const toolCosts = this.calculateToolCosts();
    const revenueImpact = this.calculateRevenueImpact(currentRevenue, growthRate);
    
    const netMonthlySavings = costSavings.monthly - toolCosts;
    const totalMonthlyBenefit = netMonthlySavings + revenueImpact.monthly;
    
    const roi = toolCosts > 0 
      ? ((totalMonthlyBenefit * 12) / (toolCosts * 12)) * 100 
      : 0;
    
    const paybackPeriod = toolCosts > 0 
      ? toolCosts / totalMonthlyBenefit 
      : 0;
    
    return {
      timeSavings: {
        hoursPerWeek: timeSavings,
        hoursPerMonth: timeSavings * 4.33
      },
      costSavings: costSavings,
      toolCosts: {
        monthly: toolCosts,
        yearly: toolCosts * 12
      },
      revenueImpact: revenueImpact,
      netBenefit: {
        monthly: totalMonthlyBenefit,
        yearly: totalMonthlyBenefit * 12
      },
      roi: Math.round(roi),
      paybackPeriod: Math.round(paybackPeriod * 10) / 10, // Months
      tools: this.tools.length
    };
  }

  generateReport() {
    const roi = this.calculateROI();
    
    return {
      summary: {
        totalTools: roi.tools,
        hoursSavedPerWeek: roi.timeSavings.hoursPerWeek,
        monthlySavings: roi.costSavings.monthly,
        monthlyRevenueIncrease: roi.revenueImpact.monthly,
        netMonthlyBenefit: roi.netBenefit.monthly,
        roi: roi.roi + '%',
        paybackPeriod: roi.paybackPeriod + ' months'
      },
      detailed: roi,
      recommendations: this.generateRecommendations(roi)
    };
  }

  generateRecommendations(roi) {
