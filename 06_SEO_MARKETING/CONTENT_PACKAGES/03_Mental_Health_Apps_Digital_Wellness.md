### Mood Tracker (Python)

"""
Mental Health Mood Tracker
Tracks daily mood, activities, and provides insights
"""

from datetime import datetime, timedelta
import json
from collections import Counter

class MoodTracker:
    def __init__(self):
        self.entries = []
        
    def log_mood(self, mood_score, activities=None, notes=None):
        """Log daily mood entry"""
        entry = {
            'date': datetime.now().isoformat(),
            'mood_score': mood_score,  # 1-10 scale
            'activities': activities or [],
            'notes': notes,
            'timestamp': datetime.now().isoformat()
        }
        self.entries.append(entry)
        return entry
    
    def get_mood_trend(self, days=30):
        """Get mood trend over specified days"""
        cutoff_date = datetime.now() - timedelta(days=days)
        recent_entries = [
            e for e in self.entries 
            if datetime.fromisoformat(e['date']) >= cutoff_date
        ]
        
        if not recent_entries:
            return None
        
        scores = [e['mood_score'] for e in recent_entries]
        avg_score = sum(scores) / len(scores)
        
        # Calculate trend
        if len(scores) >= 2:
            first_half = scores[:len(scores)//2]
            second_half = scores[len(scores)//2:]
            trend = 'improving' if sum(second_half)/len(second_half) > sum(first_half)/len(first_half) else 'declining'
        else:
            trend = 'stable'
        
        return {
            'average_mood': round(avg_score, 2),
            'entries_count': len(recent_entries),
            'trend': trend,
            'min_score': min(scores),
            'max_score': max(scores)
        }
    
    def get_activity_correlations(self):
        """Find activities correlated with better moods"""
        activity_scores = {}
        
        for entry in self.entries:
            mood = entry['mood_score']
            for activity in entry.get('activities', []):
                if activity not in activity_scores:
                    activity_scores[activity] = []
                activity_scores[activity].append(mood)
        
        correlations = {}
        for activity, scores in activity_scores.items():
            if len(scores) >= 3:  # Need at least 3 data points
                avg_mood = sum(scores) / len(scores)
                correlations[activity] = {
                    'average_mood': round(avg_mood, 2),
                    'occurrences': len(scores)
                }
        
        # Sort by average mood (highest first)
        return dict(sorted(correlations.items(), key=lambda x: x[1]['average_mood'], reverse=True))
    
    def generate_insights(self):
        """Generate insights from mood data"""
        trend = self.get_mood_trend(30)
        correlations = self.get_activity_correlations()
        
        insights = {
            'trend_analysis': trend,
            'top_activities': dict(list(correlations.items())[:5]),
            'recommendations': []
        }
        
        if trend and trend['trend'] == 'declining':
            insights['recommendations'].append('Consider trying activities that correlate with higher moods')
        
        if correlations:
            top_activity = list(correlations.keys())[0]
            insights['recommendations'].append(f'"{top_activity}" is associated with your highest moods')
        
        return insights

# Example usage
tracker = MoodTracker()

# Log daily moods
tracker.log_mood(7, ['exercise', 'meditation'], 'Feeling good today')
tracker.log_mood(6, ['work'], 'Busy day')
tracker.log_mood(8, ['exercise', 'social'], 'Great day!')
tracker.log_mood(5, ['work', 'stress'], 'Feeling overwhelmed')

# Get insights
insights = tracker.generate_insights()
print('Mood Insights:', json.dumps(insights, indent=2))
```

/**
 * Digital Wellness Check-in Bot
 * Daily wellness check-ins and recommendations
 */

class WellnessCheckIn {
  constructor() {
    this.checkIns = [];
    this.wellnessScores = {
      sleep: 0,
      mood: 0,
      stress: 0,
      exercise: 0,
      social: 0
    };
  }

  dailyCheckIn(scores) {
    const checkIn = {
      date: new Date().toISOString(),
      scores: scores,
      overall: this.calculateOverall(scores)
    };
    
    this.checkIns.push(checkIn);
    this.updateWellnessScores(scores);
    
    return {
      ...checkIn,
      recommendations: this.generateRecommendations(scores)
    };
  }

  calculateOverall(scores) {
    const total = Object.values(scores).reduce((sum, score) => sum + score, 0);
    return Math.round(total / Object.keys(scores).length);
  }

  updateWellnessScores(scores) {
    Object.keys(scores).forEach(key => {
      if (this.wellnessScores[key] !== undefined) {
        // Moving average
        this.wellnessScores[key] = (this.wellnessScores[key] * 0.7) + (scores[key] * 0.3);
      }
    });
  }

  generateRecommendations(scores) {
    const recommendations = [];
    
    if (scores.sleep < 6) {
      recommendations.push({
        category: 'sleep',
        message: 'Consider improving sleep hygiene. Try Sleep Cycle or Calm for sleep support.',
        priority: 'high'
      });
    }
    
    if (scores.mood < 6) {
      recommendations.push({
        category: 'mood',
        message: 'Your mood seems low. Consider using Headspace or talking to a therapist on BetterHelp.',
        priority: 'high'
      });
    }
    
    if (scores.stress > 7) {
      recommendations.push({
        category: 'stress',
        message: 'High stress detected. Try meditation with Insight Timer or breathing exercises.',
        priority: 'high'
      });
    }
    
    if (scores.exercise < 5) {
      recommendations.push({
        category: 'exercise',
        message: 'Regular exercise can improve mood and reduce stress. Start with 10 minutes daily.',
        priority: 'medium'
      });
    }
    
    if (scores.social < 5) {
      recommendations.push({
        category: 'social',
        message: 'Social connection is important. Consider joining a support group or reaching out to friends.',
        priority: 'medium'
      });
    }
    
    return recommendations;
  }

  getWeeklyReport() {
    const last7Days = this.checkIns.slice(-7);
    const averages = {};
    
    ['sleep', 'mood', 'stress', 'exercise', 'social'].forEach(category => {
      const values = last7Days.map(ci => ci.scores[category]).filter(v => v !== undefined);
      if (values.length > 0) {
        averages[category] = Math.round(values.reduce((a, b) => a + b, 0) / values.length);
      }
    });
    
    return {
      period: 'Last 7 days',
      averages: averages,
      trend: this.calculateTrend(last7Days),
      recommendations: this.generateWeeklyRecommendations(averages)
    };
  }

  calculateTrend(checkIns) {
    if (checkIns.length < 2) return 'insufficient_data';
    
    const firstHalf = checkIns.slice(0, Math.floor(checkIns.length / 2));
    const secondHalf = checkIns.slice(Math.floor(checkIns.length / 2));
    
    const firstAvg = firstHalf.reduce((sum, ci) => sum + ci.overall, 0) / firstHalf.length;
    const secondAvg = secondHalf.reduce((sum, ci) => sum + ci.overall, 0) / secondHalf.length;
    
    if (secondAvg > firstAvg + 0.5) return 'improving';
    if (secondAvg < firstAvg - 0.5) return 'declining';
    return 'stable';
  }

  generateWeeklyRecommendations(averages) {
    const recommendations = [];
    
    if (averages.mood < 6) {
      recommendations.push('Consider professional support through BetterHelp or Talkspace');
    }
    
    if (averages.stress > 7) {
      recommendations.push('Regular meditation with Headspace or Calm can help manage stress');
    }
    
    if (averages.sleep < 6) {
      recommendations.push('Improve sleep with Sleep Cycle or establish a bedtime routine');
    }
    
    return recommendations;
  }
}

// Example usage
const wellness = new WellnessCheckIn();

// Daily check-ins
wellness.dailyCheckIn({
  sleep: 7,
  mood: 6,
  stress: 5,
  exercise: 4,
  social: 5
});

wellness.dailyCheckIn({
  sleep: 6,
  mood: 5,
  stress: 7,
  exercise: 3,
  social: 4
});

// Get weekly report
const report = wellness.getWeeklyReport();
console.log('Weekly Wellness Report:', JSON.stringify(report, null, 2));
```
