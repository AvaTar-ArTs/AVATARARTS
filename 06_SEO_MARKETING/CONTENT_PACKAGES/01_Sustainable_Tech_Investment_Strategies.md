### Investment Calculator (Python)

```python
"""
Sustainable Tech Investment Calculator
Calculates portfolio performance and ESG impact metrics
"""

class SustainableTechPortfolio:
    def __init__(self, initial_investment, annual_contribution, years):
        self.initial = initial_investment
        self.annual = annual_contribution
        self.years = years
        
    def calculate_future_value(self, annual_return_rate):
        """Calculate future value with compound interest"""
        future_value = self.initial * (1 + annual_return_rate) ** self.years
        
        # Add annual contributions
        for year in range(1, self.years + 1):
            future_value += self.annual * (1 + annual_return_rate) ** (self.years - year)
        
        return future_value
    
    def calculate_co2_avoided(self, investment_per_ton_co2=1000):
        """Estimate CO2 avoided based on investment"""
        # Average: $1000 investment avoids 1 ton CO2 annually
        total_investment = self.initial + (self.annual * self.years)
        annual_co2_avoided = total_investment / investment_per_ton_co2
        total_co2_avoided = annual_co2_avoided * self.years
        return total_co2_avoided
    
    def generate_report(self, annual_return=0.15):
        """Generate investment report"""
        future_value = self.calculate_future_value(annual_return)
        co2_avoided = self.calculate_co2_avoided()
        total_invested = self.initial + (self.annual * self.years)
        total_return = future_value - total_invested
        
        return {
            'total_invested': total_invested,
            'future_value': future_value,
            'total_return': total_return,
            'annualized_return': annual_return,
            'co2_avoided_tons': co2_avoided,
            'years': self.years
        }

# Example usage
portfolio = SustainableTechPortfolio(
    initial_investment=10000,
    annual_contribution=5000,
    years=10
)

report = portfolio.generate_report(annual_return=0.15)
print(f"Future Value: ${report['future_value']:,.2f}")
print(f"Total Return: ${report['total_return']:,.2f}")
print(f"CO2 Avoided: {report['co2_avoided_tons']:.2f} tons")
```

### ESG Stock Screener (JavaScript)

```javascript
/**
 * ESG Stock Screener for Sustainable Tech
 * Filters stocks based on ESG criteria and financial metrics
 */

class ESGStockScreener {
  constructor(stocks) {
    this.stocks = stocks;
  }

  filterByESGScore(minScore = 70) {
    return this.stocks.filter(stock => stock.esgScore >= minScore);
  }

  filterBySector(sectors) {
    return this.stocks.filter(stock => 
      sectors.includes(stock.sector)
    );
  }

  filterByMarketCap(minCap, maxCap) {
    return this.stocks.filter(stock => 
      stock.marketCap >= minCap && stock.marketCap <= maxCap
    );
  }

  filterByGrowthRate(minGrowth) {
    return this.stocks.filter(stock => 
      stock.revenueGrowth >= minGrowth
    );
  }

  screen(options = {}) {
    let results = [...this.stocks];

    if (options.minESGScore) {
      results = this.filterByESGScore(options.minESGScore);
    }

    if (options.sectors) {
      results = this.filterBySector(options.sectors);
    }

    if (options.marketCap) {
      results = this.filterByMarketCap(
        options.marketCap.min,
        options.marketCap.max
      );
    }

    if (options.minGrowth) {
      results = this.filterByGrowthRate(options.minGrowth);
    }

    return results.sort((a, b) => b.esgScore - a.esgScore);
  }
}

// Example usage
const stocks = [
  {
    symbol: 'TSLA',
    name: 'Tesla Inc',
    esgScore: 85,
    sector: 'Clean Transportation',
    marketCap: 800000000000,
    revenueGrowth: 0.25
  },
  {
    symbol: 'FSLR',
    name: 'First Solar',
    esgScore: 92,
    sector: 'Renewable Energy',
    marketCap: 15000000000,
    revenueGrowth: 0.35
  }
];

const screener = new ESGStockScreener(stocks);
const results = screener.screen({
  minESGScore: 80,
  sectors: ['Renewable Energy', 'Clean Transportation'],
  minGrowth: 0.20
});

console.log('Filtered Stocks:', results);
```
