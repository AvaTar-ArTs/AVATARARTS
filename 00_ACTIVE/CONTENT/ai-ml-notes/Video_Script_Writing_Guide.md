import random
from collections import Counter
import re

# Sample data for hooks and CTAs
hooks = [
    "Stop scrolling! Here's why {} matters more than you think.",
    "Ever wondered about {}? This might change your perspective.",
    "If you're looking for tips on {}, you need to see this!"
]

ctas = [
    "Leave a comment if this spoke to you!",
    "Share this with someone who needs to hear it.",
    "Follow for more insights like this!"
]

# Function to generate a video script
def generate_video_script(topic, key_points):
    hook = random.choice(hooks).format(topic)
    script = [f"**Hook**: {hook}\n"]
    
    script.append("**Main Content**:")
    for i, point in enumerate(key_points, start=1):
        script.append(f"{i}. {point}")
    
    cta = random.choice(ctas)
    script.append(f"\n**Call to Action**: {cta}")
    
    return "\n".join(script)

# Function for basic text analysis to extract keywords and themes
def text_analysis(text):
    # Clean and split the text into words
    words = re.findall(r'\b\w+\b', text.lower())
    
    # Count word frequencies
    common_words = ['the', 'and', 'to', 'a', 'in', 'of', 'for', 'on', 'with', 'is', 'it', 'that']
    filtered_words = [word for word in words if word not in common_words]
    word_counts = Counter(filtered_words)
    
    # Get top 5 keywords
    keywords = [word for word, count in word_counts.most_common(5)]
    
    return {
        "total_words": len(words),
        "unique_words": len(set(words)),
        "keywords": keywords
    }

# Example key points for video content
key_points_example = [
    "Why this topic impacts daily life.",
    "Key strategies to implement.",
    "Common pitfalls to avoid.",
    "A personal story related to the theme.",
    "Final takeaways for viewers."
]

"""

# Print outputs

print("\nText Analysis Result:")
print(text_analysis_result)

        return

        if files:

            print()
