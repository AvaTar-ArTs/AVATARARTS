import pandas as pd
import re

def parse_md_to_table(md_file_path):
    """
    Reads a Markdown file and extracts rows for a structured table.
    """
    data = []
    
    with open(md_file_path, 'r') as file:
        lines = file.readlines()
    
    # Regex patterns to extract components
    trend_pattern = r"^\|\s*(\d+)\s*\|"
    title_pattern = r"\|\s*\[(.+?)\]\((.+?)\)"
    img_url_pattern = r"\|\s*(.+?)\s*\|"
    keywords_pattern = r"\|\s*(.+?)\s*$"

    for line in lines:
        if line.startswith('|') and not line.startswith('| -----'):
            trend_match = re.search(trend_pattern, line)
            title_match = re.search(title_pattern, line)
            img_url_match = re.findall(img_url_pattern, line)
            keywords_match = re.search(keywords_pattern, line)

            if trend_match and title_match and img_url_match:
                trend = trend_match.group(1)
                title_name = title_match.group(1)
                product_url = title_match.group(2)
                img_url = img_url_match[1] if len(img_url_match) > 1 else ""
                keywords = keywords_match.group(1) if keywords_match else ""

                data.append({
                    "Trend": trend,
                    "Title/Name": title_name,
                    "IMG URL": img_url,
                    "Product URL": product_url,
                    "Keywords": keywords
                })

    return pd.DataFrame(data)
