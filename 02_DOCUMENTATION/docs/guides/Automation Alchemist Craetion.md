def generate_howto_schema(title, steps, tools=None):
    """Generate HowTo schema for tutorial content"""
    schema = {
        "@context": "https://schema.org",
        "@type": "HowTo",
        "name": title,
        "step": [
            {
                "@type": "HowToStep",
                "name": step["title"],
                "text": step["description"],
                "url": f"https://aiworkflowalchemy.com{step.get('url', '')}"
            }
            for step in steps
        ]
    }
    
    if tools:
        schema["tool"] = [
            {
                "@type": "HowToTool",
                "name": tool["name"],
                "url": tool.get("url", "")
            }
            for tool in tools
        ]
    
    return schema
