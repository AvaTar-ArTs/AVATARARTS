## Automation Scripts

### Python Scripts for SEO

#### 1. Bulk URL Status Checker

```python
import requests
from urllib.parse import urlparse
import csv

def check_url_status(url):
    """Check HTTP status code for a URL"""
    try:
        response = requests.head(url, allow_redirects=True, timeout=10)
        return {
            'url': url,
            'status_code': response.status_code,
            'final_url': response.url,
            'redirects': len(response.history)
        }
    except requests.RequestException as e:
        return {
            'url': url,
            'status_code': 'Error',
            'final_url': None,
            'redirects': 0,
            'error': str(e)
        }

def check_urls_from_file(input_file, output_file):
    """Check status codes for URLs in a file"""
    with open(input_file, 'r') as f:
        urls = [line.strip() for line in f if line.strip()]
    
    results = []
    for url in urls:
        print(f"Checking: {url}")
        result = check_url_status(url)
        results.append(result)
    
    # Write results to CSV
    with open(output_file, 'w', newline='') as f:
        writer = csv.DictWriter(f, fieldnames=['url', 'status_code', 'final_url', 'redirects'])
        writer.writeheader()
        writer.writerows(results)
    
    print(f"Results saved to {output_file}")

# Usage:
# check_urls_from_file('urls.txt', 'results.csv')
```

#### 2. Sitemap URL Extractor

```python
import requests
from xml.etree import ElementTree
import csv

def extract_urls_from_sitemap(sitemap_url, output_file):
    """Extract all URLs from an XML sitemap"""
    try:
        response = requests.get(sitemap_url)
        response.raise_for_status()
        
        root = ElementTree.fromstring(response.content)
        
        # Handle namespace
        namespace = {'ns': 'http://www.sitemaps.org/schemas/sitemap/0.9'}
        
        urls = []
        for url in root.findall('.//ns:url', namespace):
            loc = url.find('ns:loc', namespace)
            lastmod = url.find('ns:lastmod', namespace)
            
            urls.append({
                'url': loc.text if loc is not None else '',
                'lastmod': lastmod.text if lastmod is not None else ''
            })
        
        # Write to CSV
        with open(output_file, 'w', newline='', encoding='utf-8') as f:
            writer = csv.DictWriter(f, fieldnames=['url', 'lastmod'])
            writer.writeheader()
            writer.writerows(urls)
        
        print(f"Extracted {len(urls)} URLs to {output_file}")
        
    except Exception as e:
        print(f"Error: {e}")

# Usage:
# extract_urls_from_sitemap('https://example.com/sitemap.xml', 'sitemap_urls.csv')
```

#### 3. Meta Tag Scraper

```python
import requests
from bs4 import BeautifulSoup
import csv

def scrape_meta_tags(url):
    """Scrape meta tags from a URL"""
    try:
        response = requests.get(url, timeout=10)
        soup = BeautifulSoup(response.content, 'html.parser')
        
        # Get title
        title = soup.find('title')
        title_text = title.string if title else ''
        
        # Get meta description
        meta_desc = soup.find('meta', attrs={'name': 'description'})
        description = meta_desc['content'] if meta_desc else ''
        
        # Get H1
        h1 = soup.find('h1')
        h1_text = h1.get_text() if h1 else ''
        
        # Get canonical
        canonical = soup.find('link', attrs={'rel': 'canonical'})
        canonical_url = canonical['href'] if canonical else ''
        
        return {
            'url': url,
            'title': title_text,
            'title_length': len(title_text),
            'description': description,
            'desc_length': len(description),
            'h1': h1_text,
            'canonical': canonical_url
        }
        
    except Exception as e:
        return {
            'url': url,
            'error': str(e)
        }

def scrape_multiple_urls(urls, output_file):
    """Scrape meta tags from multiple URLs"""
    results = []
    
    for url in urls:
        print(f"Scraping: {url}")
        result = scrape_meta_tags(url)
        results.append(result)
    
    # Write to CSV
    with open(output_file, 'w', newline='', encoding='utf-8') as f:
        fieldnames = ['url', 'title', 'title_length', 'description', 'desc_length', 'h1', 'canonical']
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        writer.writerows(results)
    
    print(f"Results saved to {output_file}")

# Usage:
# urls = ['https://example.com', 'https://example.com/about']
# scrape_multiple_urls(urls, 'meta_tags.csv')
```

function checkKeywordRankings() {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Rankings');
  var keywords = sheet.getRange('A2:A').getValues();
  var domain = sheet.getRange('B1').getValue(); // Your domain
  
  var results = [];
  
  for (var i = 0; i < keywords.length; i++) {
    if (keywords[i][0] === '') break;
    
    var keyword = keywords[i][0];
    var rank = findRanking(keyword, domain);
    results.push([new Date(), rank]);
    
    Utilities.sleep(2000); // Delay to avoid rate limiting
  }
  
  // Write results
  sheet.getRange(2, 3, results.length, 2).setValues(results);
}

  // This is a simplified example - actual implementation would need proper API

    var url = 'https://www.google.com/search?q=' + encodeURIComponent(keyword);

    }
