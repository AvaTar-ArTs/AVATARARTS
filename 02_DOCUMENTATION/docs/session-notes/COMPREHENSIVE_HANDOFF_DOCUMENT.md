# TECHNICAL DELIVERABLES

## 1. Discography Sides Parser

**File:** `/Users/steven/Music/nocTurneMeLoDieS/scripts/build_discography_sides.py`

```python
#!/usr/bin/env python3
"""
Discography Sides Generator
Converts csv/Discography ALL.csv into themed Side pages with SEO + schema
"""

import pandas as pd
import json from pathlib import Path
from collections import defaultdict
from datetime import datetime

class DiscographySidesBuilder:
    def __init__(self, csv_path="csv/Discography ALL.csv"):
        self.df = pd.read_csv(csv_path)
        self.output_dir = Path("discography_sides")
        self.output_dir.mkdir(exist_ok=True)

    def group_by_keys(self):
        """Group tracks by Keys column"""
        return self.df.groupby('Keys')

    def generate_schema(self, side_name, tracks):
        """Generate MusicAlbum schema.org JSON-LD"""
        track_list = []
        for _, track in tracks.iterrows():
            track_list.append({
                "@type": "MusicRecording",
                "name": track['Song Title'],
                "url": track['Suno URL'],
                "description": track['Analysis'][:200]  # First 200 chars
            })

        schema = {
            "@context": "https://schema.org",
            "@type": "MusicAlbum",
            "name": f"Discography Side: {side_name}",
            "byArtist": {
                "@type": "MusicGroup",
                "name": "nocTurneMeLoDieS",
                "url": "https://avatararts.org/discography"
            },
            "description": f"{len(tracks)} tracks exploring {side_name.lower()}",
            "track": track_list
        }
        return json.dumps(schema, indent=2)

    def generate_markdown(self, side_name, tracks):
        """Generate Markdown page for Side"""
        md = f"""# Discography Side: {side_name}

**Generated:** {datetime.now().strftime('%Y-%m-%d')}
**Track Count:** {len(tracks)}

---

## Overview

{self.generate_overview(side_name, tracks)}

---

## Tracks

"""
        for idx, (_, track) in enumerate(tracks.iterrows(), 1):
            md += f"""
### {idx}. {track['Song Title']}

**Analysis:** {track['Analysis']}

**Listen:** [Suno]({track['Suno URL']})

---
"""

        md += f"""
## Get the Merch

Explore lyric journals, posters, and more featuring {side_name}:
- [Etsy Shop](https://etsy.com/shop/nocturnemelodies)
- [Redbubble](https://redbubble.com/people/nocturnemelodies)

---

*Part of the nocTurneMeLoDieS Discography Sides series.*
"""
        return md

    def generate_overview(self, side_name, tracks):
        """Generate compelling overview from track analyses"""
        # Sample 3 random analyses and blend them
        samples = tracks['Analysis'].sample(min(3, len(tracks))).tolist()
        return f"A collection exploring {side_name.lower()}, blending {len(tracks)} unique perspectives into a cohesive sonic journey."

    def generate_html(self, side_name, tracks, schema):
```
