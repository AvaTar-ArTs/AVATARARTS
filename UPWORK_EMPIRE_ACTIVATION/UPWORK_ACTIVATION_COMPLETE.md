### Automation Suite Integration:
```python
# How to use with your actual Upwork data:

from upwork_automation_suite import UpworkJobAnalyzer
import pandas as pd

# Load your Upwork jobs (export from Upwork as CSV)
jobs_df = pd.read_csv('upwork_jobs.csv')

analyzer = UpworkJobAnalyzer()

# Analyze each job
for index, job in jobs_df.iterrows():
    analysis = analyzer.analyze_job_fit(
        job['title'],
        job['description']
    )

    if analysis['priority'] == 'HIGH':
        # Generate proposal
        proposal = analyzer.generate_proposal_outline(job, analysis)
        print(f"HIGH PRIORITY JOB FOUND: {job['title']}")
        print(f"Fit Score: {analysis['fit_score']}/100")
        print(f"Estimated Rate: ${analysis['estimated_rate'][0]}-{analysis['estimated_rate'][1]}/hr")
        print("\nGenerated Proposal:")
        print(proposal)
        print("\n" + "="*70 + "\n")
```
